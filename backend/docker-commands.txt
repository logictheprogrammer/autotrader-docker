---------------------
Build Node API Image
---------------------

sudo docker build -t autotrader-backend ./

---------------------
Run Node API Container
---------------------

sudo docker run --name autotrader-backend \
  --env-file ./.env \
  --rm \
  -it \
  --network host \
  -v "$(pwd)":/app \
  autotrader-backend
